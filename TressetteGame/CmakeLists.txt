cmake_minimum_required(VERSION 3.16)
project(TressetteGame LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Enable automoc
set(CMAKE_AUTOMOC ON)


set(CMAKE_PREFIX_PATH "C:/Qt/6.9.0/msvc2022_64")


# Find Qt6
find_package(Qt6 REQUIRED COMPONENTS Core Gui Qml Quick QuickControls2 QuickLayouts)

# Define source files
set(SOURCES
    TressetteGame/Cards.cpp
    TressetteGame/Deck.cpp
    TressetteGame/Game.cpp
    TressetteGame/Main.cpp
    TressetteGame/Player.cpp
    TressetteGame/GameController.cpp
)

# Define header files
set(HEADERS
    TressetteGame/Cards.hpp
    TressetteGame/Deck.hpp
    TressetteGame/Game.hpp
    TressetteGame/Player.hpp
    TressetteGame/GameController.h
)

# QML file
set(QML_FILES
    TressetteGame/Main_design.qml
)

# Qt resources
qt_add_resources(RESOURCES resources.qrc)


set(CMAKE_WINDOWS_SUBSYSTEM "WINDOWS")
# Create executable (without QML files)
add_executable(${PROJECT_NAME} ${SOURCES} ${HEADERS} ${RESOURCES})
set_target_properties(${PROJECT_NAME} PROPERTIES WIN32_EXECUTABLE ON)
# Link Qt libraries
target_link_libraries(${PROJECT_NAME} PRIVATE
    Qt6::Core
    Qt6::Gui
    Qt6::Qml
    Qt6::Quick
    Qt6::QuickControls2
    Qt6::QuickLayouts
)
# Set QML import path
qt6_import_qml_plugins(TressetteGame)

# Add QML runtime imports (Optional but helps)
set_target_properties(TressetteGame PROPERTIES
    QT_QML_IMPORT_PATH "${Qt6_DIR}/../../../qml"
)

#    QML_FILES ${QML_FILES}
#)

# Set properties for Qt integration
set_property(TARGET ${PROJECT_NAME} PROPERTY CXX_STANDARD 17)
